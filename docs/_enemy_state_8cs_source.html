98 {
99 base.OnEnter();
100 m_remainingTime = m_rushTime;

102 m_rushEffectApearProgress = 0.0f;
103

107 public override void OnUpdate()
108 {
109 base.OnUpdate();
110 m_remainingTime -= Time.deltaTime;
111 m_homingTime -= Time.deltaTime;
112
113 //ラッシュエフェクトの進捗度に合わせて表示（山本）
114 if (Core.m_rushAttackEffect)
115 {
116 //ダッシュ時間に合わせて進捗度が0=>1=>0になるように調整
117 m_rushEffectApearProgress = (float)Math.Sin((1.0f - (m_remainingTime / m_rushTime)) * Math.PI);
118 //進捗度をセット
119 Core.m_rushAttackEffect.GetComponent<VisualEffect>().SetFloat("UVAlphaClippingValue", m_rushEffectApearProgress);
120 }
121
122
123 if (m_remainingTime <= 0f)
124 {
125 Core.m_animator.SetTrigger("EndAttack");
126
127 if (Core.m_dashEffect)
128 {
129 //ダッシュエフェクトを止める（山本）
130 Core.m_dashEffect?.Stop();
131 }
132
133 if (Core.m_rushAttackEffect)
134 {
135 //ラッシュ攻撃エフェクトを非アクティブにする
136 Core.m_rushAttackEffect?.SetActive(false);
137 }
138 }
139 }

142 public override void OnFixedUpdate()
143 {
144 if (m_homingTime > 0.0f)

146 Vector3 rotVec = Core.m_inputProvider.Destination - Core.transform.position;
147 rotVec.y = 0.0f;
148 rotVec.Normalize();
149 rotVec = Core.transform.forward + rotVec * m_rateHoming;
150 Core.SetRotateToTarget(rotVec);
151 }
152 Core.Move(Core.m_dushSpeed * m_magnificationSpeed);
153 }
