129 public void UpdateRotation(ref Quaternion currentRotation, float deltaTime)
130 {
131 if (m_lookVector != Vector3.zero && OrientationSharpness > 0f)
132 {
133 m_lookVector.y = 0f;
134 m_lookVector.Normalize();
135
136 // Smoothly interpolate from current to target look direction
137 Vector3 smoothedLookInputDirection = Vector3.Slerp(Motor.CharacterForward, m_lookVector, 1 - Mathf.Exp(-OrientationSharpness * deltaTime)).normalized;
138
139 // Set the current rotation (which will be used by the KinematicCharacterMotor)

141 }

153 public void UpdateVelocity(ref Vector3 currentVelocity, float deltaTime)
154 {
155 if (Motor.GroundingStatus.IsStableOnGround)
156 {
157 // Reorient source velocity on current ground slope (this is because we don't want our smoothing to cause any velocity losses in slope changes)
158 //currentVelocity = Motor.GetDirectionTangentToSurface(currentVelocity, Motor.GroundingStatus.GroundNormal) * currentVelocity.magnitude;
159
160 // Calculate target velocity
161 //Vector3 inputRight = Vector3.Cross(_moveInputVector, Motor.CharacterUp);
162 //Vector3 reorientedInput = Vector3.Cross(Motor.GroundingStatus.GroundNormal, inputRight).normalized * _moveInputVector.magnitude;
163
164 if (!m_isRootMotion)
165 {
166 Vector3 targetMovementVelocity;
167 if (m_moveVec.magnitude == 0.0f) targetMovementVelocity = Motor.CharacterForward * m_moveSpeed * m_speedRate;
168 else targetMovementVelocity = m_moveVec * m_moveSpeed * m_speedRate;
169
170 // Smooth movement Velocity
171 currentVelocity = Vector3.Lerp(currentVelocity, targetMovementVelocity, 1 - Mathf.Exp(-StableMovementSharpness * deltaTime));
172 }
173 // ルートモーションの移動
174 else if (deltaTime > 0)
175 {
176 // The final velocity is the velocity from root motion reoriented on the ground plane
177 Vector3 Velocity = _rootMotionPositionDelta / deltaTime;
178 currentVelocity = Motor.GetDirectionTangentToSurface(Velocity, Motor.GroundingStatus.GroundNormal) * Velocity.magnitude;
179 //currentVelocity *= 3.0f;
180 }
181 }
182 else
183 {
184 if (_forwardAxis > 0f)
185 {
186 // If we want to move, add an acceleration to the velocity
187 Vector3 targetMovementVelocity = Motor.CharacterForward * _forwardAxis * m_moveSpeed * m_speedRate;
188 Vector3 velocityDiff = Vector3.ProjectOnPlane(targetMovementVelocity - currentVelocity, Gravity);
189 currentVelocity += velocityDiff * deltaTime;
190 }
191
192 // Gravity
193 currentVelocity += Gravity * deltaTime;
194 }
195
196 if (_internalVelocityAdd.sqrMagnitude > 0f)
197 {
198 currentVelocity += _internalVelocityAdd;
199 _internalVelocityAdd = Vector3.zero;
200 }
201
202 m_moveVec = Vector3.zero;
203 m_speedRate = 1.0f;
204 }
