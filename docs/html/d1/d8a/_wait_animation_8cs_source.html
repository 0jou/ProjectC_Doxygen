39 void CheckTransition()
40 {
41 Animator animator = m_animator.value as Animator;
42 if (animator == null)
43 {
44 return;
45 }
46
47 string layerName = m_layerName.value;
48 string stateName = m_stateName.value;
49
50 int layerIndex = AnimatorUtility.GetLayerIndex(animator, layerName);
51 if (layerIndex < 0)
52 {
53 Debug.LogError("指定のレイヤーが見つかりません。" + gameObject.name);
54 return;
55 }
56
57 if (animator.GetCurrentAnimatorStateInfo(layerIndex).IsName(stateName))
58 {
59 if (!animator.IsInTransition(layerIndex))
60 {
61 Transition(_NextState);
62 }
63 }
