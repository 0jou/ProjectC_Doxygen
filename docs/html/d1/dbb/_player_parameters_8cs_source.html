208 public void UpdateVanishWeapon(Animator animator)
209 {
210 if (!m_isVanishWeapon) return;
211
212 if (m_eventTime <= 0.0f)
213 {
214 m_isVanishWeapon = false;
215
216 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
217 {
218 director.Play(m_appearWeaponEvent);
219 //アニメーター側に武器を所持していることを伝える
220 animator.SetFloat("HasWeapon", 1.0f);
221 }
222 }
223 else
224 {
225 m_eventTime -= Time.deltaTime;
226 }
227 }
228
229 public void AppearWeapon(Animator animator)
230 {
231 if (!m_isVanishWeapon) return;
232 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
233 {
234 director.Play(m_appearWeaponEvent);
235 m_isVanishWeapon = false;
236 //アニメーター側に武器を所持していることを伝える
237 animator.SetFloat("HasWeapon", 1.0f);
238 }
239 }
240
241 public void HideWeapon(Animator animator)
242 {
243 if (m_isVanishWeapon) return;
244 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
245 {
246 director.Play(m_vanishWeaponEvent);
247 m_isVanishWeapon = true;
248 //アニメーター側に武器を所持してないことを伝える
249 animator.SetFloat("HasWeapon", 0.0f);
250 }
