110 virtual public bool AddItem(ItemTypeID _itemTypeID, uint _itemID)
111 {
112
113 var data = ItemDataBaseManager.instance.GetItemData(_itemTypeID, _itemID);
114 if (data == null) return false;
115
116
117 // 現在所持しているアイテムから探す
118 foreach (var list in m_itemDataList)
119 {
120 if (list.ItemTypeID != _itemTypeID || list.ItemID != _itemID) continue;
121 if (data.MaxNum <= list.Num) continue;
122
123 // 所持数を追加
124 list.Num++;
125
126 return true;
127 }
128
129 // 所持しているアイテムに追加できなければ新規作成

171

174 // 検索

177 if (data.ItemTypeID != _itemTypeID || data.ItemID != _itemID) continue;
178

181 targetData = data;
182 continue;
183 }

186 // 一番所持数が少ないものから取り除く

196
