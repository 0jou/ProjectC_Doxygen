151 private async UniTask OnShowUI(string _name)
152 {
153 //中に入ってなけらば早期リターン（山本）
154 if (m_actionDictionary.Count == 0)
155 {
156 return;
157 }
158
159 if (m_actionDictionary.ContainsKey(_name))
160 {
161 if (m_actionDictionary[_name].activeInHierarchy) return;
162
163 m_actionDictionary[_name].SetActive(true);
164 if (m_actionDictionary[_name].TryGetComponent<ActionUIBaseController>(out ActionUIBaseController controller))
165 {
166 await controller.OnShow();
167 }
168 }
169 else
170 {
171 Debug.LogError(_name + "が登録されてないです。");
172 }
173
174 }
175 private async UniTask OnHideUI(string _name)
176 {
177 //中に入ってなけらば早期リターン（山本）
178 if (m_actionDictionary.Count == 0)
179 {
180 return;
181 }
182
183 if (m_actionDictionary.ContainsKey(_name))
184 {
185 if (!m_actionDictionary[_name].activeInHierarchy) return;
186
187 if (m_actionDictionary[_name].TryGetComponent<ActionUIBaseController>(out ActionUIBaseController controller))

189 await controller.OnClose();
190 }
191 m_actionDictionary[_name].SetActive(false);
192 }
193 else
194 {
195 Debug.LogError(_name + "が登録されてないです。");

200 private async Task< AsyncOperationHandle<GameObject>> LoadPrefab(
201 string assetPath, Transform _parent,
202 List<AsyncOperationHandle<GameObject>> handles = null)// 読み込み調べるためのリスト
203 {
204 var handle = Addressables.LoadAssetAsync<GameObject>(assetPath);
205 //handle.Completed += (op) => OnGameObjectLoaded(op, _parent);
206
207 await handle.Task;
208
209 OnGameObjectLoaded(handle, _parent);
210
211 if (handles != null)

214 }
215 return handle;

218 private void OnGameObjectLoaded(AsyncOperationHandle<GameObject> handle, Transform _parent)
219 {
220 if (handle.Status == AsyncOperationStatus.Succeeded)
221 {
222 string name = handle.Result.name;
223 GameObject obj = Instantiate(handle.Result, _parent);
224 obj.SetActive(false);
225 m_actionDictionary.Add(name, obj);
226
227 if (obj.TryGetComponent<ActionUIBaseController>(out ActionUIBaseController controller))
228 {
229 controller.OnInitialize();
230 }
231
232 }
233 else
234 {
235 // Addressableでの取得に失敗した場合
236 Debug.LogError("ゲームオブジェクト取得失敗。");
237 }
238 }

240 private async void AllLoadCompleted(List<AsyncOperationHandle<GameObject>> _operationHandles)

252 SwitchState();

258 }
