133 {
134 return false;

137 // レイ判定
138 Vector3 dir = targetCollider.bounds.center - myCollider.bounds.center;
139 float dist = Vector3.Distance(targetCollider.bounds.center, myCollider.bounds.center);
140 Ray ray = new Ray(myCollider.bounds.center, dir);
141 RaycastHit hit;
142 Debug.DrawRay(ray.origin, ray.direction * dist, Color.red);
143 if (Physics.Raycast(ray, out hit, dist))
144 {
145 if (hit.collider.gameObject.name == targetCollider.gameObject.name)
146 {
147 return true;
148 }
149 }
