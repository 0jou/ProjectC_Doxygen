105 bool SearchTarget(GameObject _target, ref float _nearestDist, float _searchDist)
106 {
107 if (transform.root.name == _target.name) return false;
108
109 if (Vector3.Distance(_target.transform.position, m_startPos) >= m_chaseParameters.value.DistAwayFromSpawnPos) return false;
110
111 float dist;
112
113 dist = Vector3.Distance(transform.position, _target.transform.position);
114 if (dist <= m_chaseParameters.value.NoticeDist)
115 {
116 _nearestDist = dist;
117 return true;
118 }
119 if (dist > _searchDist) return false;
120
121 float angle;
122 angle = Vector3.Angle(transform.forward, _target.transform.position - transform.position);
123 if (angle > m_chaseParameters.value.ViewAngle * 0.5f) return false;
124
125 if (dist >= _nearestDist) return false;
126
127 Collider targetCollider;
128 if (!_target.TryGetComponent(out targetCollider))
129 {
130 return false;
131 }
132 Vector3 dir = targetCollider.bounds.center - m_myCollider.bounds.center;
133 dist = Vector3.Distance(targetCollider.bounds.center, m_myCollider.bounds.center);
134 Ray ray = new Ray(m_myCollider.bounds.center, dir);
135 RaycastHit hit;
136 Debug.DrawRay(ray.origin, ray.direction * dist, Color.red);
137 if (Physics.Raycast(ray, out hit, dist))
138 {
139 if (hit.collider.gameObject.name == targetCollider.gameObject.name)
140 {
141 _nearestDist = dist;
142 return true;
143 }
144 }
145
146 return false;
147 }
