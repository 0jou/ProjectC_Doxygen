144 public void UpdateVanishWeapon(Animator animator)
145 {
146 if (!m_isVanishWeapon) return;
147
148 if (m_eventTime <= 0.0f)
149 {
150 m_isVanishWeapon = false;
151
152 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
153 {
154 director.Play(m_appearWeaponEvent);
155 //アニメーター側に武器を所持していることを伝える
156 animator.SetFloat("HasWeapon", 1.0f);
157 }
158 }
159 else
160 {
161 m_eventTime -= Time.deltaTime;
162 }
163 }
164
165 public void AppearWeapon(Animator animator)
166 {
167 if (!m_isVanishWeapon) return;
168 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
169 {
170 director.Play(m_appearWeaponEvent);
171 m_isVanishWeapon = false;
172 //アニメーター側に武器を所持していることを伝える
173 animator.SetFloat("HasWeapon", 1.0f);
174 }
175 }
176
177 public void HideWeapon(Animator animator)

181 {

183 m_isVanishWeapon = true;
184 //アニメーター側に武器を所持してないことを伝える
185 animator.SetFloat("HasWeapon", 0.0f);
