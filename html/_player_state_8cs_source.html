458 private void CalculateThrowPower(Vector3 _startpos, Vector3 _endPos)
459 {
460 //　開始位置と到着位置の長さを計算
461 var adjacent = Vector3.Distance(
462 _startpos,
463 _endPos);
464
465 //　落下距離を計算
466 //　0.5 * 重力 * 到達時間の2乗
467 var opposite = Mathf.Abs(
468 0.5f * Physics.gravity.y * m_arrivalTime * m_arrivalTime);
469
470 //　到達点＋上向きに落下距離 放物線の頂点
471 var upPos = _endPos + Vector3.up * opposite;
472
473 //　斜辺の長さを計算
474 var hypotenuse = Vector3.Distance(
475 _startpos, upPos);
476
477 //　角度を計算
478 //　余弦定理
479 float theta = -Mathf.Acos(
480 (Mathf.Pow(hypotenuse, 2) + Mathf.Pow(adjacent, 2) - Mathf.Pow(opposite, 2))
481 / (2 * hypotenuse * adjacent));
482
483 //　横軸のspeedから斜め方向の速さを計算
484 float hypotenuseSpeed = hypotenuse / m_arrivalTime;
485
486 //　一旦攻撃対象の方を見る
487 m_lunchObj.transform.LookAt(_endPos);
489 m_lunchObj.transform.Rotate(
490 Vector3.right, theta * Mathf.Rad2Deg, Space.Self);
491
492 Core.PlayerParameters.m_throwPower =
493 m_lunchObj.transform.forward * hypotenuseSpeed;
494 }
