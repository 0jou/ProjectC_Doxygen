155 private async UniTask CreateJudgeWindow()
156 {
157 var cancelToken = this.GetCancellationTokenOnDestroy();
158
159 try
160 {
161
162 if (m_selectUIController == null)
163 {
164 Debug.LogError("SelectUIコントローラーが登録されていません");
165 return;
166 }
167
168
169 if (m_createRecipeSlotList == null)
170 {
171 Debug.LogError("CreateItemSlotListコンポーネントがアタッチされていません");
172 return;
173 }
174
175 if (m_changeRecipeDescription == null)
176 {
177 Debug.LogError("ChangeRecipeDescriptionコンポーネントがアタッチされていません");
178 return;
179 }
180
181 // 選択しているUIがなければ
182 var currentUI = m_selectUIController.CurrentSelectUI;
183 if (currentUI == null) return;
184
185 // 選択したか
186 if (!m_selectUIController.IsPress) return;
187
188 var slotData = currentUI.GetComponent<ItemSlotData>();
189 if (slotData == null) return;
190
191 CreateFoodData createFood = new();
192
193 // 料理を作成できるか確認
194 if (!createFood.IsCreate((FoodID)slotData.ItemID))
195 {
196 // Feat 作成できない場合はここにくる。作成できませんウィンドウ的なのを作りたい
197 return;
198 }
199
200 if (m_createJudgeWindow == null)
201 {
202 Debug.LogError("m_createJudgeWindowがシリアライズされていません");
203 return;
204 }
205
206 bool judge = await m_createJudgeWindow.Create<JudgeWindow>();
207 cancelToken.ThrowIfCancellationRequested();
208
209 // 作成するのであれば
210 if (judge)
211 {
212
213 createFood.CreateFood((FoodID)slotData.ItemID, true);
214
215 // 作成した料理を確認するウィンドウを作成
216 await CreateConfirmationWindowWindow(slotData.ItemTypeID, slotData.ItemID);
217 cancelToken.ThrowIfCancellationRequested();
218
219 // 作成可能か確認
220 m_createRecipeSlotList.CheckCreate();
221
222 // 説明文を更新
223 m_changeRecipeDescription.ChangeSelectUIDescription();
224
225
226 }
227
228 }
229 catch(System.OperationCanceledException ex)
230 {
231
232 }
