177 public static int GetCreateNum(PocketType _pocketType, FoodID _id)
178 {
179 var data = ItemDataBaseManager.instance.GetItemData<FoodData>(ItemTypeID.Food, (uint)_id);
180 if (data == null) return 0;
181
182 // 作成可能数
183 int num = 0;
184
185 // ループ
186 while (true)
187 {
188 // 必要な素材リスト
189 foreach (var needIngredient in data.NeedIngredientObjectList)
190 {
191 if (needIngredient == null) continue;
192
193 // 対応するポケットから素材所持数
194 var itemNum = _pocketType.GetPocketItemDataManager().GetItemNum(ItemTypeID.Ingredient, (uint)needIngredient.IngredientID);
195
196 // 現在の所持数が必要数を下回っていれば
197 if (itemNum < needIngredient.Num * (num + 1)) return num;
198 }
199
200 // 作成可能数を加算
201 num++;
202 }
203 }

209 public static int GetProvideNum(PocketType _pocketType, FoodID _id)
210 {
211 // 提供可能数
212 int num = 0;
213
214 num += _pocketType.GetPocketItemDataManager().GetItemNum(ItemTypeID.Food, (uint)_id);
215 num += GetCreateNum(_pocketType, _id);
216
217 return num;
218 }

225 public static bool RemoveNeedIngredient(PocketType _pocketType, FoodID _foodID)
226 {
227 // 作成可能でなければ取り除かない
228 if (IsCreate(_pocketType, _foodID) == false) return false;
229
230 var data = ItemDataBaseManager.instance.GetItemData<FoodData>(ItemTypeID.Food, (uint)_foodID);
231
232 // 必要な素材リスト
233 foreach (var list in data.NeedIngredientObjectList)
234 {
235 if (list == null) continue;
236
237 // 必要な素材数分
238 for (int i = 0; i < list.Num; ++i)
239 {
240 // 必要な素材を取り除く
241 _pocketType.GetPocketItemDataManager().
242 RemoveItem(ItemTypeID.Ingredient, (uint)list.IngredientID);
243 }
244 }
245
246 return true;
247 }

260 if (RemoveNeedIngredient(_pocketType, _foodID) == false)
261 {
262 return false;
263 }
264 }
265
266 return true;
