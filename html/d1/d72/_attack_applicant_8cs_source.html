72
73

75 damageable.Damaged(damageNotification, myCol, m_knockBackMultiplier);
76
77 //　当たったなら、当たった際の処理を実行
78 if (damageNotification.m_replyIsHit)
79 {
80 AddCondition(otherObj);
81 m_hittedColliders.Add(collider);
82
83 // 自分にヒットストップを適用させる
84 m_ownerInfoTag.Characore.HitStopRemainingTime = m_hitStopTime;
85
86 //BPを回復する(山本)

99
