201 public void UpdateVanishWeapon(Animator animator)
202 {
203 if (!m_isVanishWeapon) return;
204
205 if (m_eventTime <= 0.0f)
206 {
207 m_isVanishWeapon = false;
208
209 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
210 {
211 director.Play(m_appearWeaponEvent);
212 //アニメーター側に武器を所持していることを伝える
213 animator.SetFloat("HasWeapon", 1.0f);
214 }
215 }
216 else
217 {
218 m_eventTime -= Time.deltaTime;
219 }
220 }
221
222 public void AppearWeapon(Animator animator)
223 {
224 if (!m_isVanishWeapon) return;
225 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
226 {
227 director.Play(m_appearWeaponEvent);
228 m_isVanishWeapon = false;
229 //アニメーター側に武器を所持していることを伝える
230 animator.SetFloat("HasWeapon", 1.0f);
231 }
232 }
233
234 public void HideWeapon(Animator animator)
235 {
236 if (m_isVanishWeapon) return;
237 if (animator.TryGetComponent<PlayableDirector>(out PlayableDirector director))
238 {
239 director.Play(m_vanishWeaponEvent);
240 m_isVanishWeapon = true;
241 //アニメーター側に武器を所持してないことを伝える
242 animator.SetFloat("HasWeapon", 0.0f);
243 }
