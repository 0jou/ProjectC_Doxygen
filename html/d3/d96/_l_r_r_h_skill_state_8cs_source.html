149 //一番近いエネミーの方向に向き、むき終えると攻撃ステートへと移動
150 private void RotateToTargetEnemy(Vector3 targetVec)
151 {
152 if (targetVec == Vector3.zero)
153 {
154 return;
155 }
156
157 //見つめる先の位置取得
158 Vector3 nowTargetPos = Core.m_animator.GetComponent<SetRigInfo>().TargetTransform.position;
159 //目標の地点へ移動
160 nowTargetPos = Vector3.MoveTowards(nowTargetPos, m_targetPos, Core.PlayerSkillsParameters.MoveLookTargetSpeed * Time.fixedDeltaTime);
161 //座標更新
162 Core.m_animator.GetComponent<SetRigInfo>().TargetTransform.position = nowTargetPos;
163 //距離取得
164 float dist = Vector3.Distance(m_targetPos, nowTargetPos);
165
166 if (Vector3.Angle(targetVec, Core.transform.forward) < 1f && dist == 0.0f)
167 {
168 Core.m_animator.SetTrigger("SkillAttack");
169
170 return;
171 }
172
173 targetVec.Normalize();
174 //Quaternion characterTargetRotation = Quaternion.LookRotation(targetVec);
175 //Core.transform.rotation = Quaternion.RotateTowards(Core.transform.rotation, characterTargetRotation, Core.m_rotationSpeed * Time.fixedDeltaTime);
176
177 Core.m_charaCtrl.LookVector = targetVec;
178 }

181 //一番近い敵の位置情報から向く方向を取得する関数
182 private Vector3 GetNearestEnemyVector()
183 {
184 Vector3 nearestEnemyPos = Vector3.zero;
185 Vector3 targetVec = Vector3.zero;
186 float nearestDist = float.MaxValue;
187
188 foreach (var chara in IMetaAI<CharacterCore>.Instance.ObjectList)
189 {
190 if (chara.GroupNo != CharacterGroupNumber.enemy) continue;
191 if (chara.m_characterStatus.m_hp.Value <= 0.0f) continue;
192
193 if (SearchTarget(chara.gameObject, ref nearestDist))
194 {
195 nearestEnemyPos = chara.transform.position;
196 nearestEnemyPos.y += chara.GetComponent<CapsuleCollider>().center.y;
197 }
198 }
199
200 m_targetPos = nearestEnemyPos;
201
202 //Core.PlayerSkillsParameters.TargetPos = m_targetPos;
203
204 targetVec = m_targetPos - Core.transform.position;
205 targetVec.y = 0;
206 targetVec.Normalize();
207 return targetVec;
208 }

211 //メタAIから敵の情報を取ってきて、一番近いエネミーかどうかチェックする関数(壁とか考慮せず、距離だけ)
212 private bool SearchTarget(GameObject _target, ref float _nearestDist)
213 {
214 float dist;
215
216 dist = Vector3.Distance(Core.transform.position, _target.transform.position);
217
218 if (dist > _nearestDist) return false;
219
220 _nearestDist = dist;
221
222 return true;
223 }
