119 void Scroll(GameObject _ui)
120 {
121 #region nullチェック
122 if (m_scrollRect == null)
123 {
124 Debug.LogError("ScrollRectシリアライズされていません");
125 return;
126 }
127 if (m_scrollRectTransform == null)
128 {
129 Debug.LogError("RectTransformがシリアライズされていません");
130 return;
131 }
132 if (_ui == null) return;
133 #endregion
134
135 // UIの行列を取得
136 RectTransform rect;
137 if (_ui.TryGetComponent(out rect) == false) return;
138
139 // 距離
140 Vector2 dist = Vector2.zero;
141
142 switch (m_type)
143 {
144 case ChangeScrollViewPositionType.Center:
145 {
146 if (Center(rect, ref dist) == false) return;
147 break;
148 }
149
150 case ChangeScrollViewPositionType.Edge:
151 {
152 if (Edge(rect, ref dist) == false) return;
153 break;
154 }
155 }
156
157 // 現在のスクロール位置を取得
158 Vector2 currentScrollPosition = m_scrollRect.content.anchoredPosition;
159
160 // 新しいスクロール位置を計算
161 // dist.y += m_scrollRectTransform.position.y;
162 Vector2 newScrollPosition = currentScrollPosition - dist;
163
164 // 移動させる
165 DoMove(newScrollPosition);
166
167 }

216 private bool Center(RectTransform _objectTransform, ref Vector2 _dist)
217 {
218

222 return false;

228

230 Vector3 rectLocalPos = m_scrollRectTransform.InverseTransformPoint(rect.position);
231 Vector3 scrollRectLocalPos = m_scrollRectTransform.InverseTransformPoint(m_scrollRectTransform.position);

233

252 }

262 private bool Edge(RectTransform _objectTransform, ref Vector2 _dist)
263 {
264
265 if (m_scrollRectTransform == null)
266 {
267 Debug.LogError("スクロールビューのRectTransformコンポーネントがシリアライズされていません");
268 return false;
269 }
270
271 var rect = _objectTransform;
272 if (rect == null) return false;
273
274
275 // rectのローカル位置を取得
276 Vector3 rectLocalPos = m_scrollRectTransform.InverseTransformPoint(rect.position);
277 Vector3 scrollRectLocalPos = m_scrollRectTransform.InverseTransformPoint(m_scrollRectTransform.position);
278
279
280 // 変更したかどうか
281 bool isChange = false;
282
283 if (m_isVertical)
284 {
285 // rectの下端とscrollRectの下端のローカル座標を計算
286 var rectDownPos = rectLocalPos.y - (rect.rect.height / 2.0f);
287 var scrollRectDownPos = scrollRectLocalPos.y - (m_scrollRectTransform.rect.height / 2.0f);
288
289 // rectの上端とscrollRectの上端のローカル座標を計算
290 var rectUpPos = rectLocalPos.y + (rect.rect.height / 2.0f);
291 var scrollRectUpPos = scrollRectLocalPos.y + (m_scrollRectTransform.rect.height / 2.0f);
292
293 if (m_gridLayoutGroup)
294 {
295 scrollRectDownPos += m_gridLayoutGroup.padding.bottom;
296
297 scrollRectUpPos -= m_gridLayoutGroup.padding.top;
298 }
299
300 // スロットがスクロールビューの範囲外に出ている場合の処理
301 if (scrollRectUpPos < rectUpPos)
302 {
303 isChange = true;
304 _dist.y = rectUpPos - scrollRectUpPos;
305 }
306 else if (scrollRectDownPos > rectDownPos)
307 {
308 isChange = true;
309 _dist.y = rectDownPos - scrollRectDownPos;
310 }
311
312 }
313
314 if (m_isHorizontal)
315 {
316 // rectの左端とscrollRectの左端のローカル座標を計算
317 var rectLeftPos = rectLocalPos.x - (rect.rect.width / 2.0f);
318 var scrollRectLeftPos = scrollRectLocalPos.x - (m_scrollRectTransform.rect.width / 2.0f);
319
320 // rectの右端とscrollRectの左端のローカル座標を計算
321 var rectRightPos = rectLocalPos.x + (rect.rect.width / 2.0f);
322 var scrollRectRightPos = scrollRectLocalPos.x + (m_scrollRectTransform.rect.width / 2.0f);
323
324 if (m_gridLayoutGroup)
325 {
326 scrollRectLeftPos += m_gridLayoutGroup.padding.left;
327
328 scrollRectRightPos -= m_gridLayoutGroup.padding.right;
329 }
330
331 // スロットがスクロールビューの範囲外に出ている場合の処理
332 if (scrollRectRightPos < rectRightPos)
333 {
334 isChange = true;
335 _dist.x = rectRightPos - scrollRectRightPos;
336 }
337 else if (scrollRectLeftPos > rectLeftPos)
338 {
339 isChange = true;
340 _dist.x = rectLeftPos - scrollRectLeftPos;
341 }
342
343 }
344
345 return isChange;
