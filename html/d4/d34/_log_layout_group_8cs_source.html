98 private void Layout()
99 {
100
101 // 停止する
102 KillTween();
103
104 // 子オブジェクト分回す
105 for (int i = 0; i < transform.childCount; ++i)
106 {
107 Vector2 shiftPos = Vector2.zero;
108 int posNumber = i;
109
110 // 基準を決める
111 switch (m_standardType)
112 {
113 // 古いものを基準にするか
114 case StandardType.Old:
115 {
116 break;
117 }
118 // 新しいものを基準にするか
119 case StandardType.New:
120 {
121 posNumber = Mathf.Abs(posNumber - (transform.childCount - 1));
122 break;
123 }
124 }
125
126 // ずらす座標
127 shiftPos = (m_cellSize + m_spaceSize) * posNumber;
128
129 // 向きを変更
130 switch (m_directionType)
131 {
132 case DirectionType.Height:
133 {
134 shiftPos.x = 0.0f;
135 break;
136 }
137 case DirectionType.Width:
138 {
139 shiftPos.y = 0.0f;
140 break;
141 }
142 }
143
144 switch (m_shiftType)
145 {
146 // 基準UIの位置が上/左
147 case ShiftType.Top_Left:
148 {
149 shiftPos = new Vector2(Mathf.Abs(shiftPos.x), Mathf.Abs(shiftPos.y));
150 shiftPos = new Vector2(0.0f, 0.0f) - shiftPos;
151 break;
152 }
153 // 基準UIの位置が下/右
154 case ShiftType.Bottom_Right:
155 {
156 shiftPos = new Vector3(Mathf.Abs(shiftPos.x), Mathf.Abs(shiftPos.y), 0.0f);
157 break;
158 }
159 }
160
161 // DoTweenで移動
162 var obj = transform.GetChild(i);
163 if (obj.TryGetComponent<RectTransform>(out var rect))
164 {
165 var tween = rect.DOAnchorPos(shiftPos, m_duration).
166 SetEase(m_ease).SetLink(rect.gameObject);
167
168 m_tweenList.Add(tween);
169 }
170 }
171 }
