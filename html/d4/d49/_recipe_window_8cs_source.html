156 private async UniTask CreateJudgeWindow()
157 {
158 var cancelToken = this.GetCancellationTokenOnDestroy();
159
160 try
161 {
162 #region nullチェック
163
164 if (m_selectUIController == null)
165 {
166 Debug.LogError("SelectUIコントローラーが登録されていません");
167 return;
168 }
169 #endregion
170
171 // 選択したか
172 if (!m_selectUIController.IsPress) return;
173
174 // 選択しているUIがなければ
175 var currentUI = m_selectUIController.CurrentSelectUI;
176 if (currentUI == null) return;
177
178 if (currentUI.TryGetComponent<ItemSlotData>(out var slotData) == false)
179 {
180 // 選択中のUIがItemSlotDataでなければ
181 return;
182 }
183
184 #region nullチェック
185
186 if (m_createRecipeSlotList == null)
187 {
188 Debug.LogError("CreateItemSlotListコンポーネントがアタッチされていません");
189 return;
190 }
191
192 if (m_changeRecipeDescription == null)
193 {
194 Debug.LogError("ChangeRecipeDescriptionコンポーネントがアタッチされていません");
195 return;
196 }
197
198 if (m_createJudgeWindow == null)
199 {
200 Debug.LogError("m_createJudgeWindowがシリアライズされていません");
201 return;
202 }
203
204 #endregion
205
206 // 料理を作成できるか確認
207 if (FoodData.IsCreate(m_pocketType, (FoodID)slotData.ItemID) == false)
208 {
209 // 作成できなければ
210 return;
211 }
212
213 bool judge = await m_createJudgeWindow.Create<JudgeWindow>();
214 cancelToken.ThrowIfCancellationRequested();
215
216 // 作成するのであれば
217 if (judge)
218 {
219 // 料理を作成する
220 FoodData.CreateFood(m_pocketType, (FoodID)slotData.ItemID, true);
221
222 // 作成した料理を確認するウィンドウを作成
223 await CreateConfirmationWindowWindow(slotData.ItemTypeID, slotData.ItemID);
224 cancelToken.ThrowIfCancellationRequested();
225
226 // 現在作成可能か再度確認
227 m_createRecipeSlotList.CheckCreate();
228
229 // 説明文を更新
230 m_changeRecipeDescription.ChangeSelectUIDescription();
231
232 }

234 }
