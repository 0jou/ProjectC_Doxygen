565 private void CalculateThrowPower(Vector3 _startpos, Vector3 _endPos)
566 {
567 //　開始位置と到着位置の長さを計算
568 var adjacent = Vector3.Distance(
569 _startpos,
570 _endPos);
571
572 //　落下距離を計算
573 //　0.5 * 重力 * 到達時間の2乗
574 var opposite = Mathf.Abs(
575 0.5f * Physics.gravity.y * m_arrivalTime * m_arrivalTime);
576
577 //　到達点＋上向きに落下距離 放物線の頂点
578 var upPos = _endPos + Vector3.up * opposite;
579
580 //　斜辺の長さを計算
581 var hypotenuse = Vector3.Distance(
582 _startpos, upPos);
583
584 //　角度を計算
585 //　余弦定理
586 float theta = -Mathf.Acos(
587 (Mathf.Pow(hypotenuse, 2) + Mathf.Pow(adjacent, 2) - Mathf.Pow(opposite, 2))
588 / (2 * hypotenuse * adjacent));
589
590 //　横軸のspeedから斜め方向の速さを計算
591 float hypotenuseSpeed = hypotenuse / m_arrivalTime;
592
593 //　一旦攻撃対象の方を見る
594 m_lunchObj.transform.LookAt(_endPos);
596 m_lunchObj.transform.Rotate(
597 Vector3.right, theta * Mathf.Rad2Deg, Space.Self);
598
599 Core.PlayerParameters.m_throwPower =
600 m_lunchObj.transform.forward * hypotenuseSpeed;
601 }
