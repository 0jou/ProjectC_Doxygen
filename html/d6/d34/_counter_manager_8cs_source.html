144 private OrderFoodData CreateOrderFood(FoodID _foodID, CustomerData _customerData)
145 {
146 var foodData = ItemDataBaseManager.instance.GetItemData<FoodData>(ItemTypeID.Food, (uint)_foodID);
147 if (foodData == null || _customerData == null) return null;
148
149
150 // 経営用プレハブを取得/作成
151 var prefab = foodData.ManagementFoodPrefab;
152 if (prefab == null) return null;
153 var obj = Instantiate(prefab);
154 obj.SetActive(false);
155
156
157 // 注文料理データ
158 if (obj.TryGetComponent<OrderFoodData>(out var orderFoodData))
159 {
160 // 料理IDをセット
161 orderFoodData.FoodID = _foodID;
162
163 // 目的地座標をセット
164 orderFoodData.TargetTableSetData = _customerData.TargetTableSetData;
165
166 // リストに追加
167 m_orderFoodDataList.Add(orderFoodData);
168
169 // 問題なく作成出来たので、使用した食材を取り除く
170 ManagementProvideFoodManager.instance.RemoveProvideFood(_foodID);
171
172 return orderFoodData;
173 }
174 else
175 {
176 Debug.LogError("OrderFoodDataがアタッチされていません");
177 Destroy(obj);
178 return null;
179 }
180 }
