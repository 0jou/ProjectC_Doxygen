236 protected async UniTask OnDOScale(Ease _ease, float _duration, bool _fromFlg, Vector3 _endValue = default)
237 {
238 bool isUpdate = false;
239
240 // ゲームがストップ中でも動く
241 if (m_gameStopMoveType == GameStopMoveType.Enabled)
242 {
243 isUpdate = true;
244 }
245
246 if (_fromFlg == true)
247 {
248 await transform.DOScale(endValue: _endValue, duration: _duration).
249 From().
250 SetEase(_ease).
251 SetUpdate(isUpdate).
252 SetLink(gameObject);
253 }
254 else
255 {
256 await transform.DOScale(endValue: _endValue, duration: _duration).
257 SetEase(_ease).
258 SetUpdate(isUpdate).
259 SetLink(gameObject);
260 }

448 // UIの集まりを取得（吉田）
449 // TODO：スタート時に一度だけ呼び出す（今は敵などを実行中に増やすこともあるため、毎回やってる）
450 private void SetCanvasGroupList()
451 {
452 // typeで指定した型の全てのオブジェクトを配列で取得し,その要素数分繰り返す
453
454 GameObject[] allObjects = FindObjectsOfType<GameObject>(true);
455 foreach (GameObject obj in allObjects)
456 {
457
458 // 自オブジェクトとその子オブジェクトを除外する
459 if (obj == gameObject || obj.transform.parent == gameObject)
460 {
461 continue;
462 }
463
464 // シーン上に存在するオブジェクトならば処理.
465 if (!obj.activeInHierarchy) continue;
466
467 if (obj.layer != LayerMask.NameToLayer("UI")) continue;
468
469 if (obj.TryGetComponent<CanvasGroup>(out CanvasGroup canvasGroup))
470 {
471 // 既に隠れていればリストに追加しない(田内)
472 if (canvasGroup.alpha <= 0.0f) continue;
473
474 // 隠す用リストに追加
475 m_hideCanvasGroupList.Add(canvasGroup);
476 }
477 }

486 public async UniTask<WindowType> CreateToUpdateWindow<WindowType>(WindowController _windowController, bool _isSelf, System.Func<WindowType, UniTask> onBeforeInitialize = null) where WindowType : BaseWindow
487 {
488 #region nullチェック
489 if (_windowController == null)
490 {
491 Debug.LogError("引数のWindowControllerがnullです");
492 return null;
493 }
494 #endregion
495
496 var cancelToken = _windowController.GetCancellationTokenOnDestroy();
497 try
498 {
499 // 作成したウィンドウのアップデート
500 var window = await _windowController.CreateWindow<WindowType>(_isSelf, async _ =>
501 {
502 // ゲームストップ・被写界深度を停止
503 _.m_gameStopType = GameStopType.Disable;
504 _.m_depthOfFieldType = DepthOfFieldType.Disable;
505
506 // もし親ウィンドウがゲームをストップしていればストップ中でも動くようにする
507 if (m_gameStopType == GameStopType.Enabled)
508 {
509 _.m_gameStopMoveType = GameStopMoveType.Enabled;
510 }
511
512 // 追加の外部処理の実行
513 if (onBeforeInitialize != null)
514 {
515 // コンポーネントを取得し、動作を行う
516 if (_.TryGetComponent<WindowType>(out var tCom))
517 {
518 await onBeforeInitialize(tCom);
519 cancelToken.ThrowIfCancellationRequested();
520 }
521 else
522 {
523 Debug.LogError("コンポーネントが存在しませんでした。ジェネレートが合っているか確認してください");
524 }
525 }
526 });
527 cancelToken.ThrowIfCancellationRequested();
528
529 return window;
530
531 }
532 catch (System.OperationCanceledException ex)
533 {
534 Debug.Log(ex);
535 return null;
536 }
