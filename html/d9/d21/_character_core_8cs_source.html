241 // ダメージ処理
242 void IDamageable.Damaged(DamageNotification _dmgData, Collider _hitCol, float _knockBackMultiplier)
243 {
244 //ノーダメージフラグ中に早期リターン（山本）
245 if (m_isNoDamage) { return; }
246
247
248 //死亡判定とスキル等による怯み判定がバッティングした際に強制的に死亡状態へと移行させる処理（山本）
249 if (m_characterStatus.m_hp.Value <= 0.0f)
250 {
251 //死亡トリガーを逃れてしまったら強制的に死亡状態へと移行する（山本）
252 if (!m_animator.GetCurrentAnimatorStateInfo(0).IsName("Dead"))
253 {
254 m_characterStatus.m_hp.Value = 0.0f;
255 m_animator.SetTrigger("IsDead");
256 //HP0でもアタックエフェクトが表示されるようにする（山本）
257 _dmgData.m_replyIsHit = true;
258 }
259
260 return;
261 }
262
263 if (m_characterStatus.m_hp.Value > 0)
264 {
265 // ダメージ処理
266 Transform managerObj = transform.root.Find("ConditionManager");
267 if (managerObj != null && managerObj.TryGetComponent(out ConditionManager manager))
268 {
269 m_characterStatus.m_hp.Value -= _dmgData.m_damage * manager.DamageMulti();
270 }
271 else
272 {
273 m_characterStatus.m_hp.Value -= _dmgData.m_damage;
274 }
275
276 // TODO:ヒットエフェクトが出来たらここは消す
277 if (_hitCol)
278 {
279 SoundManager.Instance.Start3DPlayback("HitSoundBlow", _hitCol.transform.position);
280 }
281
282 // ヒットストップ処理
283 HitStopRemainingTime = _dmgData.m_hitStopTime;
284 // ヒット時演出
285 HitFadeBlink(Color.red);
286
287
288 }
289
290
291 // 0以下で、倒れる
292 if (m_characterStatus.m_hp.Value <= 0.0f)
293 {
294
295 m_characterStatus.m_hp.Value = 0.0f;
296 m_animator.SetTrigger("IsDead");
297 //HP0でもアタックエフェクトが表示されるようにする（山本）
298 _dmgData.m_replyIsHit = true;
299 return;
300 }
301
302
303 if (m_characterStatus.m_hp.Value > 0)
304 {
305 // ノックバック処理
306 if (_hitCol)
307 {
308 if (!m_isNoKnockBack && (m_characterStatus.m_knockBackDamage <= _dmgData.m_damage))
309 {
310 if (_knockBackMultiplier > 0.0f)
311 {
312 m_animator.SetTrigger("KnockBack");
313 m_knockBackVec = _hitCol.transform.forward;
314 m_knockBackVec.y = 0.0f;
315 m_knockBackVec.Normalize();
316 m_knockBackMultiplier = _knockBackMultiplier;
317
318 // 敵であれば索敵指示をだす
319 if (gameObject.TryGetComponent(out ArborFSM arborFSM))
320 {
321 arborFSM.SendTrigger("Damage");
322 arborFSM.parameterContainer.SetVector3("AttackedVec", -m_knockBackVec);
323 }
324 }
325 }
326 }
327
328 // 返信用データ用意
329 _dmgData.m_replyIsHit = true;
330 }
331 }
