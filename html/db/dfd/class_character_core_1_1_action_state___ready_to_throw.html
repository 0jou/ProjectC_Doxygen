569 //　開始位置と到着位置の長さを計算
570 var adjacent = Vector3.Distance(
571 _startpos,
572 _endPos);
573
574 //　落下距離を計算
575 //　0.5 * 重力 * 到達時間の2乗
576 var opposite = Mathf.Abs(
577 0.5f * Physics.gravity.y * m_arrivalTime * m_arrivalTime);
578
579 //　到達点＋上向きに落下距離 放物線の頂点
580 var upPos = _endPos + Vector3.up * opposite;
581
582 //　斜辺の長さを計算
583 var hypotenuse = Vector3.Distance(
584 _startpos, upPos);
585
586 //　角度を計算
587 //　余弦定理
588 float theta = -Mathf.Acos(
589 (Mathf.Pow(hypotenuse, 2) + Mathf.Pow(adjacent, 2) - Mathf.Pow(opposite, 2))
590 / (2 * hypotenuse * adjacent));
591
592 //　横軸のspeedから斜め方向の速さを計算
593 float hypotenuseSpeed = hypotenuse / m_arrivalTime;
