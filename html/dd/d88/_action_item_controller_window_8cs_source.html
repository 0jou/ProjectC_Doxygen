244 private void SelectItemAction()
245 {
246 if (m_createSelectUseItemWindow == null)
247 {
248 Debug.LogError("m_createJudgeWindowがシリアライズされていません");
249 return;
250 }
251
252 if (m_selectUIController == null)
253 {
254 Debug.LogError("SelectUIコントローラーが登録されていません");
255 return;
256 }
257
258
259 //UIリストとスロットが空ならリターン
260 if ((m_selectUIController.UIList.Count == 0) || (m_createSlotList.SlotList.Count == 0))
261 {
262 return;
263 }
264
265 // 選択しているUIがなければリターン
266 if (!m_selectUIController.CurrentSelectUI)
267 {
268 return;
269 }
270
271 // 選択しているUIがスロットになければリターン
272 var slotData = m_selectUIController.CurrentSelectUI.GetComponent<ItemSlotData>();
273 if (slotData == null)
274 {
275 return;
276 }
277
278 // プレイヤーへアイテム情報を送る処理(吉田)
279 GameObject player = GameObject.FindGameObjectWithTag("Player");
280 if (player == null)
281 {
282 return;
283 }
284
285 if (player.TryGetComponent(out CharacterCore characterCore))
286 {
287
288 var animator = characterCore.m_animator;
289
290 if (animator == null)
291 {
292 return;
293 }
294
295
296 if (characterCore.PlayerParameters == null)
297 {
298 return;
299 }
300
301 //選択したアイテム情報をプレイヤーに渡す
302 characterCore.PlayerParameters.SetPutItemInfo(slotData.ItemTypeID, slotData.ItemID);
303
304
305
306 //料理を置く
307 if (m_selectUIController.IsPut)
308 {
309 if (!animator.GetBool("IsPutItem"))
310 {
311 animator.SetBool("IsPutItem", true);
312 //武器所持状態なら武器を消失
313 player.GetComponent<CharacterCore>().PlayerParameters.HideWeapon(animator);
314 return;
315 }
316 }
317
318 //料理を置く
319 if (m_selectUIController.IsUse)
320 {
321 var data = ItemDataBaseManager.instance.GetItemData<FoodData>(slotData.ItemTypeID, slotData.ItemID);
322
323 if (data == null)
324 {
325 return;
326 }
327
328 //食べられるなら食べる状態へ移行
329 if (data.IsEat && !animator.GetBool("IsEatItem"))
330 {
331 animator.SetBool("IsEatItem", true);
332 //武器所持状態なら武器を消失
333 player.GetComponent<CharacterCore>().PlayerParameters.HideWeapon(animator);
334 return;
335 }
336 else
337 {
338 return;
339 }
340
341 }
342
343 //料理を投げる
344 if (m_selectUIController.IsThrow)
345 {
346
347 if (!animator.GetBool("IsThrow"))
348 {
349 animator.SetBool("IsThrow", true);
350 //武器所持状態なら武器を消失
351 player.GetComponent<CharacterCore>().PlayerParameters.HideWeapon(animator);
352
353 return;
354 }
