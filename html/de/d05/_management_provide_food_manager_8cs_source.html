163 public bool AddProvideFoodList(FoodID _foodID)
164 {
165 // 追加できるか確認する
166 if (IsAddList() == false)
167 {
168 return false;
169 }
170
171 // 既に追加されていれば追加しない
172 if (IsAddedProvideFood(_foodID))
173 {
174 return false;
175 }
176
177 // 作成
178 ProvideFoodData provideFoodData = new();
179 provideFoodData.FoodID = _foodID;
180 provideFoodData.ProvideNum = (uint)FoodData.GetProvideNum(m_pocketType, _foodID);
181
182 // リストに追加
183 m_provideFoodDataList.Add(provideFoodData);
184
185 // リスト変更イベントを送信
186 PublishChangeProvideFoodEvent();
187
188 // 追加イベントを送信
189 PublishAddListEvent(provideFoodData);
190
191
192 return true;
193 }

232 public void RemoveProvideFood(FoodID _id)
233 {
234 foreach (var data in m_provideFoodDataList)
235 {
236 if (data == null) continue;
237 if (data.FoodID != _id) continue;
238
239 // ボーナスがあれば食材を使用しない
240 if (0 < data.BonusNum)
241 {
242 data.BonusNum--;
243 return;
244 }
245
246 // 通常
247 if (FoodData.RemoveProvideNeedIngredient(m_pocketType, _id))
248 {
249 // 作成可能数を減算
250 data.ProvideNum--;
251
252 // 取り除いた数を加算
253 data.RemoveNum++;
254
255 // 取り除きイベントの送信
256 PublishRemoveListEvent(data);
257
258 // 売り切れイベントの確認・送信
259 PublishSoldOutProvideFoodEvent(data);
260
261 // リスト変更イベントを送信
262 PublishChangeProvideFoodEvent();
263 }
264 break;
265 }
