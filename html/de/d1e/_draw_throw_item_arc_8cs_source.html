199 private float GetArcHitTime(float startTime, float endTime)
200 {
201 // Linecastする線分の始終点の座標
202 Vector3 startPosition = GetArcPositionAtTime(startTime);
203 Vector3 endPosition = GetArcPositionAtTime(endTime);
204
205 // 衝突判定
206 RaycastHit hitInfo;
207 if (Physics.Linecast(startPosition, endPosition, out hitInfo))
208 {
209 // 衝突したColliderまでの距離から実際の衝突時間を算出
210 float distance = Vector3.Distance(startPosition, endPosition);
211 return startTime + (endTime - startTime) * (hitInfo.distance / distance);
212 }
213 return float.MaxValue;
214 }
