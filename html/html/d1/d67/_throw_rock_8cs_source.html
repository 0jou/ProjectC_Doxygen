75 Vector3 CalcThrowPower(Vector3 _startPos, Vector3 _endPos)
76 {
77 //　開始位置と到着位置の長さを計算
78 var adjacent = Vector3.Distance(
79 _startPos,
80 _endPos);
81
82 float angle = Vector3.SignedAngle(m_lunchObj.root.forward, _endPos - _startPos, Vector3.up);
83 if (Mathf.Abs(angle) > m_throwAngle)
84 {
85 Vector3 toEndVec = _endPos - _startPos;
86 toEndVec.Normalize();
87 if (angle > 0)
88 _endPos = Quaternion.Euler(0, 0, m_throwAngle * Mathf.Rad2Deg) * toEndVec + _startPos;
89 else
90 _endPos = Quaternion.Euler(0, 0, -m_throwAngle * Mathf.Rad2Deg) * toEndVec + _startPos;
91 }
92
93
94 //　落下距離を計算
95 //　0.5 * 重力 * 到達時間の2乗
96 var opposite = Mathf.Abs(
97 0.5f * Physics.gravity.y * m_arrivalTime * m_arrivalTime);
98
99 //　到達点＋上向きに落下距離 放物線の頂点
100 var upPos = _endPos + Vector3.up * opposite;
101
102 //　斜辺の長さを計算
103 var hypotenuse = Vector3.Distance(
104 _startPos, upPos);
105
106 //　角度を計算
107 //　余弦定理
108 float theta = -Mathf.Acos(
109 (Mathf.Pow(hypotenuse, 2) + Mathf.Pow(adjacent, 2) - Mathf.Pow(opposite, 2))
110 / (2 * hypotenuse * adjacent));
111
112 //　横軸のspeedから斜め方向の速さを計算
113 float hypotenuseSpeed = hypotenuse / m_arrivalTime;
114
115 // 一旦攻撃対象の方を見る
116 m_lunchObj.LookAt(_endPos);
117 //　砲台の高さ向きを変える
118 m_lunchObj.Rotate(Vector3.right, theta * Mathf.Rad2Deg, Space.Self);
119
120 //Vector3 result = Quaternion.Euler(0, 0, -90) * toEndVec;
121 //result = Quaternion.AngleAxis(theta * Mathf.Rad2Deg, Vector3.right) * result;
122
124 //transform.Rotate(Vector3.right, theta * Mathf.Rad2Deg, Space.Self);
125
126 return m_lunchObj.forward * hypotenuseSpeed;
127 }
