153 {
154 #region nullチェック

157 Debug.LogError("SelectUIControllerがシリアライズされていません");

168 return;
169 }

171
172 var cancelToken = this.GetCancellationTokenOnDestroy();
173
174 try
175 {
176 if (m_selectUIController.IsPress == false) return;
177
178 var currentUI = m_selectUIController.CurrentSelectUI;
179 if (currentUI == null || currentUI.TryGetComponent<RecipeItemSlotData>(out var slotData) == false) return;
180
181 // 料理を作成できるか確認
182 if (slotData.IsCreate == false) return;
183
184
185 // 料理作成ウィンドウ
186 var createFoodController = Instantiate(m_createFoodWindowController);
187 var window = await CreateToUpdateWindow<CreateFoodWindow>(createFoodController, true, async _ =>
188 {
189 _.SetFoodData(m_pocketType, (FoodID)slotData.ItemID);
190 await UniTask.CompletedTask;
191 });
192
193 bool createFlg = false;
194 if (window != null)
195 {
196 createFlg = await window.OnSelfUpdate();
197 await window.OnClose();
198 await window.OnDestroy();
199 }
200 if (createFoodController != null) Destroy(createFoodController.gameObject);
201
202
203 // 料理を作成すれば
204 if (createFlg)
205 {
206 // アイテム確認ウィンドウ
207 var confirmationItemController = Instantiate(m_confirmationItemWindowController);
208 await CreateToUpdateWindow<ConfirmationItemWindow>(confirmationItemController, false, async _ =>
209 {
210 _.SetDescription(m_pocketType, slotData.ItemTypeID, slotData.ItemID);
211 await UniTask.CompletedTask;
212 });
213 if (confirmationItemController != null) Destroy(confirmationItemController.gameObject);
214
215
216 // レシピ情報を初期化
217 InitializeRecipeData();
218 }
219
