154 private OrderFoodData CreateOrderFood(FoodID _foodID, CustomerData _customerData)
155 {
156 // 料理データを取得
157 var foodData = ItemDataBaseManager.instance.GetItemData<FoodData>(ItemTypeID.Food, (uint)_foodID);
158 if (foodData == null) return null;
159
160 // プレハブ
161 var prefab = foodData.ManagementFoodPrefab;
162 if (prefab == null)
163 {
164 Debug.LogError("プレハブが存在しないため作成できませんでした");
165 return null;
166 }
167
168 //====================
169 // インスタンスを作成
170
171 var obj = Instantiate(prefab);
172 obj.SetActive(false);
173
174 // 注文料理データ
175 var orderFoodData = obj.GetComponentInChildren<OrderFoodData>();
176 if (orderFoodData == null)
177 {
178 Debug.LogError("OrderFoodDataがアタッチされていません");
179 Destroy(obj);
180 return null;
181 }
182 // 料理IDをセット
183 orderFoodData.FoodID = _foodID;
184
185 // 目的地座標をセット
186 if (_customerData)
187 {
188 orderFoodData.TargetTableSetData = _customerData.TargetTableSetData;
189 }
190
191 // リストに追加
192 AddOrderFoodDataList(orderFoodData);
193
194
195 return orderFoodData;
