172 // キャンバスグループリストを取得する
173 private List<CanvasGroup> GetCanvasGroupList()
174 {
175 List<CanvasGroup> canvasGroups = new();
176
177 foreach (GameObject obj in FindObjectsOfType(typeof(GameObject)))
178 {
179 // シーン上に存在するオブジェクトならば処理.
180 if (!obj.activeInHierarchy) continue;
181
182 if (obj.layer != LayerMask.NameToLayer("UI")) continue;
183
184 if (obj.TryGetComponent<CanvasGroup>(out CanvasGroup canvasGroup))
185 {
186 canvasGroups.Add(canvasGroup);
187 }
188 }
189
190 return canvasGroups;
